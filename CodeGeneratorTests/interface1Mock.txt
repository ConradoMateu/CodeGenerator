class PokemonMock: Pokemon {
    var canEvolve: Bool!

    var eatFood: String?
    var drinkWater: String?

    var shouldCallSuccessHandler = true
    var shouldCallFailureHandler = true

    var failureHandlerParam0: String?
    
    var eatFoodWasCalled: Bool?
    var drinkWasCalled: Bool?
    var isFlyingWasCalled: Bool?
    var evolveSuccessHandlerWasCalled: Bool?

    var isFlyingShouldReturn: Bool!

    func eat(food: String) {
        eatFood = food
        eatFoodWasCalled = true
    }

    func drink(_ water: String) {
        drinkWater = water
        drinkWasCalled = true
    }

    func isFlying() -> Bool {
        isFlyingWasCalled = true
        return isFlyingShouldReturn
    }

    func evolve(successHandler: @escaping () -> Void, failureHandler: (String?) -> String?) {
        if shouldCallSuccessHandler {
            successHandler()
        }
        if shouldCallFailureHandler {
            _ = failureHandler(failureHandlerParam0)
        }
        evolveSuccessHandlerWasCalled = true
    }

}
